package com.acme.billing.persistence;

import java.time.LocalDate;
import java.math.BigDecimal;
import java.util.stream.Stream;

import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Modifying;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;

public interface CustomerRepository extends JpaRepository<Customer, Long> {

    // stmt-014: DECLARE C-ACTIVE CURSOR FOR
    //   SELECT * FROM BILLDB.CUSTOMER WHERE STATUS = 'A' AND OPENED_ON >= :WS-SINCE
    @Query("select c from Customer c where c.status = 'A' and c.openedOn >= :since")
    Stream<Customer> findActiveSince(@Param("since") LocalDate since);

    // stmt-022: UPDATE BILLDB.CUSTOMER SET BALANCE = :WS-BALANCE WHERE CUST_ID = :WS-CUST-ID
    @Modifying
    @Query("update Customer c set c.balance = :balance where c.custId = :custId")
    int updateBalance(@Param("custId") long custId,
                      @Param("balance") BigDecimal balance);
}
