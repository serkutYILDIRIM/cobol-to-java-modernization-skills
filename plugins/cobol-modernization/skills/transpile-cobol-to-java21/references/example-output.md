package com.acme.billing.legacy.bill010;

import com.acme.billing.dto.BillRecord;
import com.acme.billing.legacy.bill010.exception.DomainValidationException;
import com.acme.billing.persistence.Customer;
import com.acme.billing.persistence.CustomerRepository;
import java.math.BigDecimal;
import java.math.MathContext;
import java.math.RoundingMode;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@Transactional
public class Bill010Service {

    private static final Logger log = LoggerFactory.getLogger(Bill010Service.class);
    private static final BigDecimal CHARGE_RATE = new BigDecimal("0.015");

    private final CustomerRepository customerRepository;

    public Bill010Service(CustomerRepository customerRepository) {
        this.customerRepository = customerRepository;
    }

    /** Entry point — lowered from PROCEDURE DIVISION USING LK-BILL-INPUT. */
    public void run(BillRecord input) {
        var status = validateCustomer(input);          // 1000-VALIDATE-CUSTOMER
        switch (status) {
            case ValidateCustomerOutcome.Active a       -> calcCharges(input);
            case ValidateCustomerOutcome.Suspended s    -> log.info("BILL010: customer {} suspended ({})",
                                                                    input.custId(), s.reason());
        }
    }

    /* 1000-VALIDATE-CUSTOMER ------------------------------------------------ */
    ValidateCustomerOutcome validateCustomer(BillRecord input) {
        // BR-BILLING-001 (VALIDATION)
        if (input.custId() == null || input.custId().isBlank()) {
            return new ValidateCustomerOutcome.Suspended("CUST-ID blank");
        }
        /* end BR-BILLING-001 */

        // BR-BILLING-002 (VALIDATION)
        if (input.balance().signum() < 0) {
            return new ValidateCustomerOutcome.Suspended("balance < 0");
        }
        /* end BR-BILLING-002 */

        // WHEN OTHER
        return new ValidateCustomerOutcome.Active();
    }

    /* 2000-CALC-CHARGES ----------------------------------------------------- */
    void calcCharges(BillRecord input) {
        // BR-BILLING-003 (CALCULATION)
        var charge = input.balance()
                          .multiply(CHARGE_RATE, MathContext.DECIMAL64)
                          .setScale(2, RoundingMode.HALF_UP);
        /* end BR-BILLING-003 */

        // BR-BILLING-004 (IO-EFFECT)
        // EXEC SQL UPDATE BILLDB.CUSTOMER SET LAST_CHARGE = :WS-CHARGE
        //                                  WHERE CUST_ID = :LK-CUST-ID
        var customer = customerRepository.findById(input.custId())
                .orElseThrow(() -> new DomainValidationException(
                        "BILL010", "customer not found: " + input.custId()));
        customer.setLastCharge(charge);
        customerRepository.save(customer);
        /* end BR-BILLING-004 */
    }
}
